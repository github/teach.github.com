	# Stash your pending changes
	git stash

	# List your stashes
	git stash list

	# Stash your pending changes
	git stash save “<Message>”

	# Merge & delete the latest stash
	git stash pop

	# Merge & delete a stash
	git stash pop stash@{0}

	# Merge & keep the latest stash
	git stash apply
