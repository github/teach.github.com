	# Stash your pending changes
	git stash

	# List your stashes
	git stash list

	# Merge & delete the latest stash
	git stash pop

	# Merge & delete a stash
	git stash pop <name>

	# Remove a stash
	git stash drop <name>

	# Merge & keep the latest stash
	git stash apply
